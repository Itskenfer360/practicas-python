643def estadisticasTupla(t):
644    """
645    Recibe una tupla de números y devuelve una tupla con:
646    (mínimo, máximo, media)
647    """
648    if len(t) == 0:
649        return None
650
651    minimo = t[0]
652    maximo = t[0]
653    suma = 0
654
655    for x in t:
656        if x < minimo:
657            minimo = x
658        if x > maximo:
659            maximo = x
660        suma = suma + x
661
662    media = suma / len(t)
663    return (minimo, maximo, media)

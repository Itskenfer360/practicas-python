135    def __init__(self, x, y):
136        self.x = x
137        self.y = y
138
139        # Propiedades por defecto 
140        self.color = "#7FFF00"
141        self.grosor = 2.0
142        self.estilo_linea = "-"   # '-', '--', '-.', ':'
143        self.marcador = None      # 'o', 'x', '^', etc.

145    def set_estilo(self, color=None, grosor=None, estilo_linea=None, marcador=None):
146        if color is not None:

148        if grosor is not None:
149            self.grosor = grosor
150        if estilo_linea is not None:
151            self.estilo_linea = estilo_linea
152        if marcador is not None:
153            self.marcador = marcador
154
155    def plot_basico(self, titulo="Gráfica"):
156        plt.figure()

161            linestyle=self.estilo_linea,

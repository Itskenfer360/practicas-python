192def resolverSistema():
193    """Ejercicio 7
194    Resuelve un sistema de 3 ecuaciones con 3 incognitas.
195    Obtiene la matriz del sistema, su determinante, la matriz inversa,
196    el vector de terminos independientes y la solucion final.
197    Mi sistema propuesto
198    Sea el siguiente sistema de ecuaciones con tres incognitas x, y, z:
199
200        x +  y +  z =  6
201        2x -  y +  z =  3
202        x + 0y + 2z =  7
203
204    Entrada 
205    A: Matriz de coeficientes del sistema establecido
206    B: Matriz de terminos independientes del sistema 
207    """
208
209    # Matriz de coeficientes
210    A = np.array([
211        [1,  1, 1],
212        [2, -1,  1],
213        [1, 0,  2]
214    ])
215
216    # Vector de terminos independientes
217    b = np.array([6, 3, 7])
218
219    print("Matriz del sistema A:")
220    print(A)
221
222    # Determinante
223    detA = np.linalg.det(A)
224    print("\nDeterminante de A:", detA)
225
226    # Matriz inversa
227    if detA != 0:
228        A_inv = np.linalg.inv(A)
229        print("\nMatriz inversa de A:")
230        print(A_inv)
231    else:
232        print("\nEl sistema no tiene matriz inversa (determinante 0)")
233
234    print("\nVector de terminos independientes b:")
235    print(b)
236
237    # Solucion del sistema
238    x = np.linalg.solve(A, b)
239    print("\nSolucion del sistema (x, y, z):")
240    print(x)

194    """Ejercicio 7
195    Resuelve un sistema de 3 ecuaciones con 3 incognitas.
196    Obtiene la matriz del sistema, su determinante, la matriz inversa,
197    el vector de terminos independientes y la solucion final.
198    Mi sistema propuesto
199    Sea el siguiente sistema de ecuaciones con tres incognitas x, y, z:
200
201        x +  y +  z =  6
202        2x -  y +  z =  3
203        x + 0y + 2z =  7
204
205    Entrada 
206    A: Matriz de coeficientes del sistema establecido
207    B: Matriz de terminos independientes del sistema 
208    """
209
210    # Matriz de coeficientes
211    A = np.array([
212        [1,  1, 1],
213        [2, -1,  1],
214        [1, 0,  2]
215    ])
216
217    # Vector de terminos independientes
218    b = np.array([6, 3, 7])
219
220    print("Matriz del sistema A:")
221    print(A)
222
223    # Determinante
224    detA = np.linalg.det(A)
225    print("\nDeterminante de A:", detA)
226
227    # Matriz inversa
228    if detA != 0:
229        A_inv = np.linalg.inv(A)
230        print("\nMatriz inversa de A:")
231        print(A_inv)
232    else:
233        print("\nEl sistema no tiene matriz inversa (determinante 0)")
234
235    print("\nVector de terminos independientes b:")
236    print(b)
237
238    # Solucion del sistema
239    x = np.linalg.solve(A, b)
240    print("\nSolucion del sistema (x, y, z):")
241    print(x)
